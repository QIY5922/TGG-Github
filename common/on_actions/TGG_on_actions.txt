on_actions = {
	on_startup = {
		effect = {
			set_variable = { TGG_monthly_tick_dirty = 0 }
		}
		effect = {
			every_country = {
				SetGNP = yes
			}
		}
	}
	on_monthly = {
		effect = {
			add_to_variable = { TGG_monthly_tick_dirty = 1 }
		}
		effect = {
			every_country = {
				SetGNP = yes
			}
		}
	}
}
